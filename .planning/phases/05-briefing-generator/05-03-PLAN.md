    from fpdf import FPDF

    FONT_PATH = "/app/fonts/DejaVuSans.ttf"

    def render_pdf(briefing: dict) -> bytes:
        pdf = FPDF()
        pdf.add_page()
        pdf.add_font("DejaVu", fname=FONT_PATH)
        pdf.set_font("DejaVu", size=16)
        pdf.cell(text="Threat Intelligence Briefing", new_x="LMARGIN", new_y="NEXT")
        pdf.set_font("DejaVu", size=11)
        pdf.cell(
            text=f"Period: last {briefing['period_hours']}h  |  Generated: {briefing['created_at']}",
            new_x="LMARGIN",
            new_y="NEXT",
        )
        pdf.ln(8)
        pdf.multi_cell(w=0, text=briefing["text"])
        return bytes(pdf.output())

    Key constraints:
    - FONT_PATH at module level (not inside render_pdf) — lets tests monkeypatch pdf_renderer.FONT_PATH
    - multi_cell(w=0, ...) is idiomatic fpdf2 for full-page-width word-wrap (RESEARCH.md Pattern 5 note)
    - bytes(pdf.output()) — pdf.output() returns bytearray in fpdf2 2.x; wrapping in bytes() ensures type is bytes
    - No error handling in render_pdf() itself — FileNotFoundError propagates to caller; generator.py's run_generate catches it and sets status: "error"
